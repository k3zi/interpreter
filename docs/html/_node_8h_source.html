39 // This macro allows us to simplify the definitions of the nodes for the CORE
40 // abstract syntax tree.
41 #define DEFINE_NODE(CLASS, PRIVATE_MEMBERS, DESTRUCTION) \
42  class CLASS : public Node { \
43  /* The staring location of the first token parsed by this node. */ \
44  Token Tok; \
45  \
46  PRIVATE_MEMBERS \
47  \
48  public: \
49  CLASS(Parser *P, class StmtSeq *SeqContext, bool SplitsContext); \
50  CLASS(Parser *P, class StmtSeq *SeqContext) \
51  : CLASS(P, SeqContext, false){}; \
52  \
53  /* \
54  * \brief Whether the node can parse against the grammar at the current \
55  * token. \
56  * \
57  * \discussion By keeping a seperate `canParse` method we are essemtialy \
58  * supporting DRY and information hiding. The more general alternative \
59  * would be to test if the current token is of a kind that our node \
60  * supports parsing (or any of the starting nonterminals in one of our \
61  * node's production rules supports parsing); this is essemtialy what \
62  * `canParse` does but it keeps associated code together. A node that \
63  * parses a <fac> only has to know that it can parse a <fac>. \
64  * Parsing a <fac> is defined in `Fac::canParse` as the conjuction of \
65  * being able to parse an <id>, <int>, or ( <expr> ). The advantage is \
66  * that the implementer of a node does not have to worry about what it's \
67  * starting nonterminals look like. They only have to pay attenion to \
68  * their production rules. Also if this parser ever expanded and a lower \
69  * leaf aquired a new production rule the changes wouldn't have to \
70  * propogate up. \
71  */ \
72  static bool canParse(Parser *P); \
73  \
74  /* \
75  * Print the given no with the passed in indetntation to the end of the \
76  * stream. \
77  */ \
78  void Print(std::ostringstream &X, unsigned Indent) override; \
79  void Execute(ASTContext &C) override; \
80  Token getToken() const { return Tok; } \
81  virtual ~CLASS(){DESTRUCTION}; \
82  };
